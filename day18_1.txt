	<input type="number" id="marks" placeholder="Enter marks">
	<button onclick="getGrade()">Check Grade</button>
	<p id="result"></p>
	<script>
		function getGrade()
		{
			let m=document.getElementById("marks").value;
			let g= m>=90?"A":
				   m>=80?"B":
				   m>=70?"C":
				   m>=60?"D": "F";
				   document.getElementById("result").innerText="Grade:"+g;
		}
	</script>
