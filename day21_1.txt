<body>

  <div id="container">
    <img id="animatedImage" src="car.jpg" alt="Animated" style="width: 30%;">
  </div>

  <button onclick="startAnimation()">Start</button>
  <button onclick="stopAnimation()">Stop</button>

  <script>
    let img = document.getElementById("animatedImage");
    let position = 0;
    let animationId;

    function moveImage() {
      if (position < window.innerWidth - img.width) {
        position += 2;
        img.style.left = position + "px";
        animationId = requestAnimationFrame(moveImage);
      }
    }

    function startAnimation() {
      if (!animationId) {
        animationId = requestAnimationFrame(moveImage);
      }
    }

    function stopAnimation() {
      cancelAnimationFrame(animationId);
      animationId = null;
    }
  </script>
